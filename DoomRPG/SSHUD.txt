// SS2 HUD (WIP)
statusbar normal, fullscreenoffsets
{
	// [Bottom left]
	// [RC] Main hud display, most critical info
	ininventory NOT "DRPGShield"
	{
		drawimage "HUDMAIN", 0, -49;
	}
	else
	{
		drawimage "HUDMAIN2", 0, -49;
	}
	
	// [RC] Bars for the most important info.
	// Health
	ininventory "DRPGShield"
	{
		drawbar "HUDHBar", "HUDNBar", health, horizontal, interpolate(4096), 38, -41;
		drawnumber 4, SMALLFONT, red, globalarray(52), alignment(right), Interpolate(4096), 142, -40;
	}
	else
	{
		drawbar "HUDHBar", "HUDNBar", health, horizontal, interpolate(4096), 38, -41;
		drawnumber 4, SMALLFONT, red, health, alignment(right), Interpolate(4096), 142, -40;
	}
	
	// EP
	drawbar "HUDEBar", "HUDNBar", DRPGEP, horizontal, interpolate(4096), 38, -30;
	drawnumber 4, SMALLFONT, lightblue, globalarray(2), alignment(right), Interpolate(4096), 142, -29;

	// Armor
	ininventory "DRPGShield"
	{
		drawbar "Shield2", "Shield", DRPGShieldCapacity, vertical, Interpolate(4096), 3, -40;
		drawnumber 5, SMALLFONT, cyan, globalarray(50), alignment(center), drawshadow, Interpolate(4096), 19, -17;
	}
	else
	{
		drawimage armoricon, 19, -24, centerbottom;
		drawnumber 4, SMALLFONT, green, armor, whennotzero, alignment(right), Interpolate(4096), 31, -28;
	}
	
	// [Bottom right]
	// Ammo
	drawimage "HUDAMMO", -37, -65;
	drawimage ammoicon1, -18, -31, centerbottom, 64, 64, alternateonfail
	drawimage ammoicon2, -18, -31, centerbottom;
	drawnumber 4, SMALLFONT, white, ammo1, alignment(center), Interpolate(4096), -18, -25;
	drawnumber 4, SMALLFONT, white, ammo1capacity, alignment(center), Interpolate(4096), -18, -12;

	// [RC] Currently selected inventory
	drawimage "HUDINV", -80, -31;

	// Key Bar
	drawkeybar 12, vertical, reverserows, auto, -50, -62, 0, 3, 12;
	
	// Current Inventory Item
	drawselectedinventory center, itemflash, SMALLFONT, -58, -13, -42, -12, white;
	
	// [RC] Return of the Jed- er, Credits!
	ininventory DRPGCredits
		drawnumber 10, SSFONT, gold, globalarray(18), alignment(right), Interpolate(409600000), 134, -10;

	// Powerup displays
	ininventory PowerIronFeet
	{
		drawimage "SUITA0", -96, 32, center;
		DrawNumber 4, SMALLFONT, white, poweruptime("RadSuit"), drawshadow, alignment(center), -96, 28;
	}
	ininventory PowerInvulnerable
		ininventory not DRPGPowerStimIndestructible
		{
			drawimage "PINVA0", -64, 16, center;
			DrawNumber 4, SMALLFONT, gold, poweruptime("InvulnerabilitySphere"), drawshadow, alignment(center), -64, 12;
			DrawNumber 4, SMALLFONT, green, poweruptime("DRPGInvulnerabilityChargePower"), drawshadow, alignment(center), -64, 12;
		}
	ininventory PowerInvisibility
	{
		drawimage "PINSA0", -32, 16, center;	
		DrawNumber 4, SMALLFONT, lightblue, poweruptime("BlurSphere"), drawshadow, alignment(center), -32, 12;
	}
	ininventory PowerShadow
	ininventory not "DRPGPowerStimShadow"
	{
		drawimage "PINSA0", -32, 16, center;
		DrawNumber 4, SMALLFONT, lightblue, poweruptime("DRPGInvisibilityChargePower"), drawshadow, alignment(center), -32, 12;
	}
	ininventory PowerStrength
		drawimage "PSTRA0", -64, 48, center;
	ininventory PowerScanner
		drawimage "PMAPA0", -32, 48, center;
	ininventory PowerLightAmp
	{
		drawimage "PVISA0", -96, 80, center;
		DrawNumber 4, SMALLFONT, brick, poweruptime("Infrared"), drawshadow, alignment(center), -96, 76;
	}
	ininventory "DRPGContinue"
	{
		drawimage "CONTA0", -30, 80, center;
		DrawNumber 4, SMALLFONT, gold, globalarray(19), drawshadow, alignment(center), -32, 76;
	}
	ininventory PowerTimeFreezer
		ininventory not "DRPGPowerStimChrono"
		{
			drawimage "TIMEA0", -96, 112, center;
			DrawNumber 4, SMALLFONT, green, poweruptime("DRPGTimeFreezer"), drawshadow, alignment(center), -96, 108;
		}
	
	// Regeneration Boost Timer
	ininventory "DRPGRegenBoost"
		ininventory not "DRPGPowerStimAdrenaline"
		{
			drawimage "REGNA0", -64, 112, center;
			DrawNumber 4, SMALLFONT, purple, globalarray(26), drawshadow, alignment(center), -64, 108;
		}
}
